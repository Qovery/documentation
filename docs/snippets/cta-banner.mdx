export const CTABanner = ({ title, subtitle, button = 'Create free account →', href = 'https://console.qovery.com' }) => {
  const [variant, setVariant] = useState('control')

  useEffect(() => {
    function readVariant() {
      const key = Object.keys(localStorage).find(k => k.startsWith('ph_phc_') && k.endsWith('_posthog'))
      if (!key) return null
      try {
        const data = JSON.parse(localStorage.getItem(key)) || {}
        return (data.$enabled_feature_flags || {})['docs-cta-subtitle-v2'] || null
      } catch (e) { return null }
    }

    const v = readVariant()
    if (v) { setVariant(v); return }

    let attempts = 0
    const t = setInterval(() => {
      const v = readVariant()
      if (v || ++attempts > 30) { clearInterval(t); if (v) setVariant(v) }
    }, 100)

    return () => clearInterval(t)
  }, [])

  function trackClick() {
    const key = Object.keys(localStorage).find(k => k.startsWith('ph_phc_') && k.endsWith('_posthog'))
    if (!key) return
    try {
      const data = JSON.parse(localStorage.getItem(key)) || {}
      if (!data.distinct_id) return
      fetch('https://phprox.qovery.com/capture/', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        keepalive: true,
        body: JSON.stringify({
          api_key: key.slice(3, -8),
          event: 'docs_cta_click',
          distinct_id: data.distinct_id,
          properties: { variant, page: location.pathname, $current_url: location.href }
        })
      })
    } catch (e) {}
  }

  return (
    <div className="cta-banner">
      <div className="cta-banner-text">
        <h3>{title}</h3>
        {variant !== 'no-subtitle' && <p className="cta-banner-subtitle">{subtitle}</p>}
      </div>
      <a href={href} className="cta-banner-button" onClick={trackClick}>{button}</a>
    </div>
  )
}
